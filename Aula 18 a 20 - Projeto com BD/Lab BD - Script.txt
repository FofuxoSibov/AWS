USE labdb;

CREATE TABLE tb_tiposusuario(
id_tipousuario INT(2),
titulo VARCHAR(255),
PRIMARY KEY(id_tipousuario)
);

CREATE TABLE tb_usuarios(
id_usuario INT(5),
nome VARCHAR(255),
cpf VARCHAR(11),
id_tipousuario INT(2),
PRIMARY KEY(id_usuario),
FOREIGN KEY(id_tipousuario) REFERENCES tb_tiposusuario(id_tipousuario)
);
